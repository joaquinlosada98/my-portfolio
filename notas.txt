query MyQuery {
    allFile(
        filter: {extension: {regex: "/(jpg)|(jpeg)|(png)/"}, relativeDirectory: {eq: "dev"}}
    ) {
        edges {
        node {
            id
            name
            absolutePath
            childImageSharp {
                gatsbyImageData(width: 60)
                }
        }
        }
    }
}
